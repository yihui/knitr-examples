\begin{document}



Create link texts in the data:

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\color{fgcolor}\begin{kframe}
\begin{alltt}
\hlkwd{library}\hlstd{(lattice)}
\hlkwd{data}\hlstd{(iris)}
\hlstd{iris}\hlopt{$}\hlstd{hyperlink} \hlkwb{<-} \hlkwd{paste}\hlstd{(}\hlstr{"\textbackslash{}\textbackslash{}hyperlink\{"}\hlstd{, iris}\hlopt{$}\hlstd{Species,} \hlstr{"\}\{"}\hlstd{, iris}\hlopt{$}\hlstd{Species,}
    \hlstr{"\}"}\hlstd{,} \hlkwc{sep} \hlstd{=} \hlstr{""}\hlstd{)}
\hlstd{iris}\hlopt{$}\hlstd{hypertarget} \hlkwb{<-} \hlkwd{paste}\hlstd{(}\hlstr{"\textbackslash{}\textbackslash{}hypertarget\{"}\hlstd{, iris}\hlopt{$}\hlstd{Species,} \hlstr{"\}\{"}\hlstd{, iris}\hlopt{$}\hlstd{Species,}
    \hlstr{"\}"}\hlstd{,} \hlkwc{sep} \hlstd{=} \hlstr{""}\hlstd{)}
\end{alltt}
\end{kframe}
\end{knitrout}


Draw these texts in plots (you can click on the labels):

\begin{knitrout}
\definecolor{shadecolor}{rgb}{0.969, 0.969, 0.969}\begin{kframe}
\begin{alltt}
\hlkwd{xyplot}\hlstd{(Sepal.Length} \hlopt{~} \hlstd{Sepal.Width} \hlopt{|} \hlstd{hyperlink,} \hlkwc{data} \hlstd{= iris)}
\end{alltt}
\end{kframe}
